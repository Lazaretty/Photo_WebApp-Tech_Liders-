create table [Photo]
(
	Id int IDENTITY(1,1),
	AuthorId int,
	[Name] nvarchar(255),
	[CameraName] nvarchar(255),
	[ShootingParameters] nvarchar(255),
	[Category] nvarchar(255),
	[Photo] varbinary(MAX)
)

create table Album (
	Id int IDENTITY(1,1),
	UserId int,
	[Name] nvarchar(255),
	)

create table PhotosInAlbum (
	AlbumId int,
	PhotoId int,
	)

	create table [User](
	ID int IDENTITY(1,1) Primary key,
	Email nvarchar(255),
	Password nvarchar(255),
	NickName nvarchar(255)
);
